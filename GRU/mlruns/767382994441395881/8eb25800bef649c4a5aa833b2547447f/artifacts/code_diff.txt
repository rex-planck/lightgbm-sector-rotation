+import qlib
+from qlib.constant import REG_CN
+from qlib.utils import init_instance_by_config
+from qlib.workflow import R
+# 🔥 核心变化：使用最基础的 DataHandler，它最听话
+from qlib.data.dataset.handler import DataHandler
+import pandas as pd
+import os
+
+# 1. 初始化
+provider_uri = r"E:\Quant_program\Qlib-Cache\cn_data"
+qlib.init(provider_uri=provider_uri, region=REG_CN)
+
+# 2. 定义自定义因子
+custom_factors = [
+    "( $close - Mean($close, 5) ) / Mean($close, 5)",  # 5日乖离
+    "$volume / Mean($volume, 20)",  # 量比
+    "Std($close, 20) / $close",  # 波动率
+    "($open - Ref($close, 1)) / Ref($close, 1)",  # 隔夜跳空
+    "($close - $open) / $open"  # 日内力度
+]
+
+
+# 🔥🔥🔥 核心修复：参数清洗类 🔥🔥🔥
+class MyAlphaHandler(DataHandler):
+    def __init__(self, instruments, start_time, end_time, data_loader, **kwargs):
+        # 1. 外科手术：强行把导致报错的时间参数剔除
+        # 这样父类 DataHandler 就不会因为收到"不认识的参数"而报错了
+        kwargs.pop('fit_start_time', None)
+        kwargs.pop('fit_end_time', None)
+
+        # 2. 调用父类
+        super().__init__(instruments, start_time, end_time, data_loader, **kwargs)
+
+
+# 3. 配置工作流
+market = "csi300"
+benchmark = "SH000300"
+
+# 定义时间常量，方便下面多次使用
+FIT_START = "2018-01-01"
+FIT_END = "2020-12-31"
+
+data_handler_config = {
+    "start_time": "2018-01-01",
+    "end_time": "2022-12-31",
+
+    # 这里的参数是留给 MyAlphaHandler 过滤用的，防止最外层初始化报错
+    "fit_start_time": FIT_START,
+    "fit_end_time": FIT_END,
+
+    "instruments": market,
+
+    # 核心：使用 data_loader 加载自定义因子
+    "data_loader": {
+        "class": "QlibDataLoader",
+        "module_path": "qlib.data.dataset.loader",
+        "kwargs": {
+            "config": {
+                "feature": custom_factors,
+                "label": ["Ref($close, -5) / $close - 1"],
+            },
+        },
+    },
+
+    # 🔥🔥🔥 核心修复：定点投喂 🔥🔥🔥
+    # 我们直接把时间写在处理器的 kwargs 里，精确控制，不再依赖自动传递
+    "infer_processors": [
+        {
+            "class": "RobustZScoreNorm",
+            "kwargs": {
+                "fields_group": "feature",
+                "clip_outlier": True,
+                "fit_start_time": FIT_START,  # 👈 显式传入
+                "fit_end_time": FIT_END  # 👈 显式传入
+            }
+        },
+        {"class": "Fillna", "kwargs": {"fields_group": "feature"}},
+    ],
+    "learn_processors": [
+        {"class": "DropnaLabel"},
+        {
+            "class": "CSRankNorm",
+            "kwargs": {
+                "fields_group": "label",
+                "fit_start_time": FIT_START,  # 👈 显式传入
+                "fit_end_time": FIT_END  # 👈 显式传入
+            }
+        },
+    ],
+}
+
+task = {
+    "model": {
+        "class": "LGBModel",
+        "module_path": "qlib.contrib.model.gbdt",
+        "kwargs": {
+            "loss": "mse",
+            "colsample_bytree": 0.8879,
+            "learning_rate": 0.0421,
+            "subsample": 0.8789,
+            "lambda_l1": 205.6999,
+            "lambda_l2": 580.9768,
+            "max_depth": 8,
+            "num_leaves": 210,
+            "num_threads": 20,
+        },
+    },
+    "dataset": {
+        "class": "DatasetH",
+        "module_path": "qlib.data.dataset",
+        "kwargs": {
+            "handler": {
+                "class": "MyAlphaHandler",
+                "module_path": "__main__",
+                "kwargs": data_handler_config,
+            },
+            "segments": {
+                "train": (FIT_START, FIT_END),
+                "valid": ("2021-01-01", "2021-12-31"),
+                "test": ("2022-01-01", "2022-12-31"),
+            },
+        },
+    },
+}
+
+if __name__ == "__main__":
+    with R.start(experiment_name="research_custom_factors"):
+        print("🧪 正在进行自定义因子实验 (Silver Bullet)...")

+

+        model = init_instance_by_config(task["model"])

+
+        print("2. 开始训练 (LightGBM)...")
+        model.fit(dataset)
+
+        # 预测与IC分析
